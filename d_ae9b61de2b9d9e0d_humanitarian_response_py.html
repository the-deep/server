179 def get_content(self, url, params): 

180 resp = requests.get(url, params={}) 

181 return resp.text 

182 

183 def fetch(self, params, offset=None, limit=None): 

184 results = [] 

185 url = self.URL 

186 if params.get('country'): 

187 url = self.URL + '/locations/' + params['country'] 

188 content = self.get_content(url, {}) 

189 soup = Soup(content, 'html.parser') 

190 contents = soup.find('div', {'id': 'content'}).find('tbody') 

191 for row in contents.findAll('tr'): 

192 try: 

193 tds = row.findAll('td') 

194 title = tds[0].find('a').get_text().strip() 

195 datestr = tds[3].get_text().strip() 

196 date = datetime.strptime(datestr, '%m/%d/%Y') 

197 url = tds[4].find('a')['href'] 

198 data = { 

199 'id': url, 

200 'title': title, 

201 'published_on': date.date(), 

202 'url': url, 

203 'source': 'Humanitarian Response', 

204 'author': 'Humanitarian Response', 

205 'website': self.URL, 

206 'source_type': Lead.SourceType.WEBSITE 

207 } 

208 results.append(data) 

209 except Exception as e: 

210 logger.warning( 

211 "Exception parsing humanitarian response connector: " + 

212 str(e.args) 

213 ) 

214 

215 final_results = results[offset: offset + limit] 

216 return final_results, len(results)
