88def _tabular_meta_extract_geo(geodata): 

89 field = geodata.field 

90 project = field.sheet.book.project 

91 project_geoareas = GeoArea.objects.filter( 

92 admin_level__region__project=project 

93 ) 

94 geodata_data = [] 

95 

96 is_code = field.get_option('geo_type', 'name') == 'code' 

97 admin_level = field.get_option('admin_level') 

98 if admin_level: 

99 project_geoareas = project_geoareas.filter( 

100 admin_level__level=admin_level 

101 ) 

102 

103 for row in geodata.field.sheet.data: 

104 similar_areas = [] 

105 query = row.get(str(field.pk)) 

106 

107 if is_code: 

108 geoareas = project_geoareas.filter(code=query).annotate( 

109 similarity=models.Value(1, models.FloatField()), 

110 ) 

111 else: 

112 geoareas = project_geoareas.annotate( 

113 similarity=search.TrigramSimilarity('title', query), 

114 ).filter(similarity__gt=0.2).order_by('-similarity') 

115 

116 for geoarea in geoareas: 

117 similar_areas.append({ 

118 'id': geoarea.pk, 

119 'similarity': geoarea.similarity, 

120 }) 

121 geodata_data.append({ 

122 'similar_areas': similar_areas, 

123 'selected_id': geoareas.first().pk if geoareas.exists() else None, 

124 }) 

125 geodata.data = geodata_data 

126 geodata.save() 

127 return True
