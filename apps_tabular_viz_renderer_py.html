81def calc_data(field): 

82 val_column = get_val_column(field) 

83 

84 data, df = clean_real_data(field.actual_data, val_column) 

85 

86 if data.empty: 

87 logger.warning('Empty DataFrame: no numeric data to calculate for field ({})'.format(field.pk)) 

88 return [], {} 

89 

90 if val_column not in data.columns: 

91 logger.warning('{} not present in field ({})'.format(val_column, field.pk)) 

92 return None, {} 

93 

94 data = data.groupby(val_column).count()['empty'].sort_values().to_frame() 

95 data = data.rename(columns={'empty': 'count', val_column: 'value'}) 

96 

97 data['value'] = data.index 

98 health_stats = { 

99 'empty': int(df[df['empty'] == True]['empty'].count()), # noqa 

100 'invalid': int(df[df['invalid'] == True]['invalid'].count()), # noqa 

101 'total': len(df.index), 

102 } 

103 return data.to_dict(orient='records'), health_stats

106def generate_chart(field, chart_type, images_format=['svg']): 

107 params = { 

108 'x_label': field.title, 

109 'y_label': 'count', 

110 'x_params': {}, 

111 'chart_size': (8, 4), 

112 'format': images_format, 

113 # data will be added according to chart type 

114 } 

115 

116 if chart_type not in [HISTOGRAM, WORDCLOUD]: 

117 df = pd.DataFrame(field.cache.get('series')) 

118 if df.empty or 'value' not in df.columns: 

119 return None 

120 params['data'] = df 

121 if field.type == Field.STRING: # NOTE: revered is used for ascending order 

122 params['x_params']['autorange'] = 'reversed' 

123 params['data']['value'] = params['data']['value'].str.slice(0, 30) + '...' # Pre slice with ellipses 

124 elif field.type == Field.GEO: 

125 adjust_df = pd.DataFrame([ 

126 {'value': 0, 'count': 0}, # Count 0 is min's max value 

127 {'value': 0, 'count': 5}, # Count 5 is max's min value 

128 ]) 

129 params['data'] = params['data'].append(adjust_df, ignore_index=True) 

130 elif field.type == Field.DATETIME: 

131 if df['value'].count() > 10: 

132 params['x_params']['tickformat'] = '%d-%m-%Y' 

133 else: 

134 params['x_params']['type'] = 'category' 

135 params['x_params']['ticktext'] = [ 

136 datetime.strptime(value, '%Y-%m-%dT%H:%M:%S').strftime('%d-%m-%Y') 

137 for value in df['value'] 

138 ] 

139 params['x_params']['tickvals'] = df['value'] 

140 

141 else: 

142 val_column = get_val_column(field) 

143 df, _ = clean_real_data(field.actual_data, val_column) 

144 if chart_type == HISTOGRAM: 

145 params['data'] = pd.to_numeric(df[val_column]) 

146 elif chart_type == WORDCLOUD: 

147 params['data'] = ' '.join(df[val_column].values) 

148 

149 if isinstance(params['data'], pd.DataFrame) and params['data'].empty: 

150 logger.warning('Empty DataFrame: no numeric data to plot for field ({})'.format(field.pk)) 

151 return None 

152 

153 chart_render = CHART_RENDER.get(chart_type) 

154 if chart_render: 

155 images = chart_render(**params) 

156 return images 

157 return None
