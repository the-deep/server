41def get_top_entity_contributor(project, Entity): 

42 contributors = ProjectMembership.objects.filter( 

43 project=project, 

44 ).annotate( 

45 entity_count=models.functions.Coalesce(models.Subquery( 

46 Entity.objects.filter( 

47 project=project, 

48 created_by=models.OuterRef('member'), 

49 ).order_by().values('project') 

50 .annotate(cnt=models.Count('*')).values('cnt')[:1], 

51 output_field=models.IntegerField(), 

52 ), 0), 

53 ).order_by('-entity_count').select_related('member', 'member__profile')[:5] 

54 

55 return [ 

56 { 

57 'id': contributor.id, 

58 'name': contributor.member.profile.get_display_name(), 

59 'user_id': contributor.member.id, 

60 'count': contributor.entity_count, 

61 } for contributor in contributors
