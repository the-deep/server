for field in sheet.field_set.all():

sheet.data_row_index = 2 

sheet.save() 

 

# check if field actual_data changed or not

# check if Re-triggered or not 

assert field.cache['status'] == Field.CACHE_PENDING

assert len(field.actual_data) == 9
