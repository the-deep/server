41def _insert_element_before(parent, elm, successors): 

42 """ 

43 Insert *elm* as child of *parent* before any existing child having 

44 tag name found in *successors*. 

45 """ 

46 successor = _first_child_found_in(parent, successors) 

47 if successor is not None: 

48 successor.addprevious(elm) 

49 else: 

50 parent.append(elm) 

51 return elm
