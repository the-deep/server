32def _first_child_found_in(parent, tagnames): 

33 """ 

34 Return the first child of parent with tag in *tagnames*, or None if 

35 not found. 

36 """ 

37 for tagname in tagnames: 

38 child = parent.find(qn(tagname)) 

39 if child is not None: 

40 return child 

41 return None

44def _insert_element_before(parent, elm, successors): 

45 """ 

46 Insert *elm* as child of *parent* before any existing child having 

47 tag name found in *successors*. 

48 """ 

49 successor = _first_child_found_in(parent, successors) 

50 if successor is not None: 

51 successor.addprevious(elm) 

52 else: 

53 parent.append(elm) 

54 return elm
