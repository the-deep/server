265 def load_text_from_text_widgets(self, entries, text_widget_ids): 

266 """ 

267 Prefetch entry texts from text_Widgets 

268 text_widget_ids: [id1, id2, [id1, 'widget-key', 'widget-label']] 

269 """ 

270 # User defined widget order (Generate order map) 

271 widget_map = { 

272 id: index 

273 for index, id in enumerate(text_widget_ids) if isinstance(id, int) 

274 } 

275 conditional_widget_map = { 

276 f'{array[0]}-{array[1]}': (index, array[2]) 

277 for index, array in enumerate(text_widget_ids) if isinstance(array, list) and len(array) == 3 

278 } 

279 

280 # Collect text from textWidgets for given entries using user defined widget data 

281 collected_widget_text = {} 

282 for attribute_d in Attribute.objects.filter( 

283 entry__in=entries, 

284 data__value__isnull=False, 

285 widget__id__in=[(id if isinstance(id, int) else id[0]) for id in text_widget_ids], 

286 ).values( 

287 'entry_id', 'data__value', 'widget__title', 'widget__id', 'widget__widget_id' 

288 ): 

289 entry_id = attribute_d['entry_id'] 

290 widget_id = attribute_d['widget__id'] 

291 widget_type = attribute_d['widget__widget_id'] 

292 

293 if widget_type == 'conditionalWidget': 

294 selected_widget_key = attribute_d['data__value'].get('selected_widget_key') 

295 widget_order, widget_title = ( 

296 conditional_widget_map.get(f'{widget_id}-{selected_widget_key}', (None, None)) 

297 ) 

298 text = ( 

299 ( 

300 attribute_d['data__value'].get(selected_widget_key) or {} 

301 ).get('data') or {} 

302 ).get('value') 

303 else: 

304 widget_order = widget_map[widget_id] 

305 widget_title = attribute_d['widget__title'] 

306 text = attribute_d['data__value'] 

307 

308 if widget_order is None: 

309 continue 

310 

311 # Map text to entry->order->text 

312 collected_widget_text[entry_id] = collected_widget_text.get(entry_id, {}) 

313 collected_widget_text[entry_id][widget_order] = [ 

314 widget_title, 

315 text and text.strip(), # Widget Text Value 

316 ] 

317 

318 self.collected_widget_text = collected_widget_text 

319 return self

359 def _get_geo_admin_level_1_data(self, geo_id_values): 

360 if len(geo_id_values) == 0: 

361 return 

362 

363 render_values = [] 

364 for region_id, admin_levels in self.region_data.items(): 

365 for admin_level in admin_levels: 

366 geo_area_titles = admin_level['geo_area_titles'] 

367 for geo_id in geo_id_values: 

368 if geo_id not in geo_area_titles: 

369 continue 

370 if geo_id in self.geoarea_data_cache: 

371 title = self.geoarea_data_cache[geo_id] 

372 title and render_values.append(title) 

373 continue 

374 self.geoarea_data_cache[geo_id] = None 

375 

376 title = geo_area_titles[geo_id].get('title') 

377 parent_id = geo_area_titles[geo_id].get('parent_id') 

378 if admin_level['level'] == 1: 

379 title and render_values.append(title) 

380 self.geoarea_data_cache[geo_id] = title 

381 continue 

382 

383 # Try to look through parent 

384 for _level in range(0, admin_level['level'] - 1)[::-1]: 

385 if parent_id: 

386 _geo_area_titles = admin_levels[_level]['geo_area_titles'] 

387 _geo_area = _geo_area_titles.get(parent_id) or {} 

388 _title = _geo_area.get('title') 

389 parent_id = _geo_area.get('parent_id') 

390 if _level == 1: 

391 _title and render_values.append(_title) 

392 self.geoarea_data_cache[geo_id] = title 

393 break 

394 

395 if render_values: 

396 return INTERNAL_SEPARATOR.join(set(render_values))

607 def _load_into_levels( 

608 self, 

609 entry, 

610 keys, 

611 levels, 

612 entries_map, 

613 valid_levels, 

614 ): 

615 """ 

616 Map an entry into corresponding levels 

617 """ 

618 parent_level_valid = False 

619 for level in levels: 

620 level_id = level.get('id') 

621 valid_level = (level_id in keys) 

622 

623 if valid_level: 

624 if not entries_map.get(level_id): 

625 entries_map[level_id] = [] 

626 entries_map[level_id].append(entry) 

627 

628 sublevels = level.get('sublevels') 

629 if sublevels: 

630 valid_level = valid_level or self._load_into_levels( 

631 entry, 

632 keys, 

633 sublevels, 

634 entries_map, 

635 valid_levels, 

636 ) 

637 

638 if valid_level and level_id not in valid_levels: 

639 valid_levels.append(level_id) 

640 

641 parent_level_valid = parent_level_valid or valid_level 

642 return parent_level_valid

644 def _generate_for_levels( 

645 self, 

646 levels, 

647 level_entries_map, 

648 valid_levels, 

649 structures=None, 

650 heading_level=2, 

651 ): 

652 """ 

653 Generate paragraphs for all entries in this level and recursively 

654 do it for further sublevels 

655 """ 

656 if structures is not None: 

657 level_map = dict((level.get('id'), level) for level in levels) 

658 levels = [level_map[s['id']] for s in structures] 

659 

660 for level in levels: 

661 if level.get('id') not in valid_levels: 

662 continue 

663 

664 title = level.get('title') 

665 entries = level_entries_map.get(level.get('id')) 

666 sublevels = level.get('sublevels') 

667 

668 if entries or sublevels: 

669 self.doc.add_heading(title, heading_level) 

670 self.doc.add_paragraph() 

671 

672 if entries: 

673 iterable_entries = entries[:Export.PREVIEW_ENTRY_SIZE] if self.is_preview else entries 

674 [self._generate_for_entry(entry) for entry in iterable_entries] 

675 

676 if sublevels: 

677 substructures = None 

678 if structures: 

679 substructures = next(( 

680 s.get('levels') for s in structures 

681 if s['id'] == level.get('id') 

682 ), None) 

683 

684 self._generate_for_levels( 

685 sublevels, 

686 level_entries_map, 

687 valid_levels, 

688 substructures, 

689 heading_level + 1, 

690 )
