59 def load_text_from_text_widgets(self, entries, text_widget_ids):

61 Prefetch entry texts from text_Widgets 

62 text_widget_ids: [id1, id2, [id1, 'widget-key', 'widget-label']]

65 widget_map = { 

66 id: index 

67 for index, id in enumerate(text_widget_ids) if isinstance(id, int)

69 conditional_widget_map = {

72 }

76 for attribute_d in Attribute.objects.filter(

79 widget__id__in=[(id if isinstance(id, int) else id[0]) for id in text_widget_ids],

81 'entry_id', 'data__value', 'widget__title', 'widget__id', 'widget__widget_id'

92 text = ( 

93 ( 

94 attribute_d['data__value'].get(selected_widget_key) or {} 

95 ).get('data') or {}
