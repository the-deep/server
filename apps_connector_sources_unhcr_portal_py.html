214class UNHCRPortal(Source): 

215 URL = 'https://data2.unhcr.org/en/search' 

216 title = 'UNHCR Portal' 

217 key = 'unhcr-portal' 

218 options = [ 

219 { 

220 'key': 'country',

222 'title': 'Country',

224 }, 

225 { 

226 'key': 'date_from', 

227 'field_type': 'date', 

228 'title': 'From', 

229 }, 

230 { 

231 'key': 'date_to', 

232 'field_type': 'date', 

233 'title': 'To', 

234 }, 

235 { 

236 'key': 'page', 

237 'field_type': 'number', 

238 'title': 'Page', 

239 }, 

240 ] 

241 params = { 

242 'type[]': [ 

243 'document', 

244 # NOTE: for now have only documents as results, other do not seem 

245 # to be parsable 

246 # 'link', 'news', 'highlight' 

247 ] 

248 } 

249 

250 def get_content(self, url, params): 

251 resp = requests.get(self.URL, params=params) 

252 return resp.text 

253 

254 def fetch(self, params, offset, limit): 

255 results = [] 

256 if params.get('country'): 

257 params['country_json'] = '{"0":"' + params['country'] + '"}' 

258 params.update(self.params) # type is default 

259 

260 content = self.get_content(self.URL, params) 

261 soup = Soup(content, 'html.parser') 

262 contents = soup.findAll('ul', {'class': 'searchResults'}) 

263 if not contents: 

264 return results, 0 

265 

266 content = contents[0] 

267 items = content.findAll('li', {'class': ['searchResultItem']}) 

268 

269 total_count = len(items) 

270 limited_items = items[offset: offset + limit] 

271 

272 for item in limited_items: 

273 itemcontent = item.find( 

274 'div', 

275 {'class': ['searchResultItem_content', 'media_body']} 

276 ) 

277 urlcontent = item.find( 

278 'div', 

279 {'class': 'searchResultItem_download'} 

280 ) 

281 datecontent = item.find( 

282 'span', 

283 {'class': 'searchResultItem_date'} 

284 ) 

285 title = itemcontent.find('a').get_text() 

286 pdfurl = urlcontent.find('a')['href'] 

287 raw_date = datecontent.find('b').get_text() # 4 July 2018 

288 date = datetime.datetime.strptime(raw_date, '%d %B %Y') 

289 data = { 

290 'title': title.strip(), 

291 'published_on': date.date(), 

292 'url': pdfurl, 

293 'source': 'UNHCR Portal', 

294 'author': 'UNHCR Portal', 

295 'source_type': '', 

296 'website': 'data2.unhcr.org' 

297 } 

298 results.append(data) 

299 

300 return results, total_count
