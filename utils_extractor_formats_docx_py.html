55def qn(tag): 

56 """ 

57 Stands for 'qualified name', a utility function to turn a namespace 

58 prefixed tag name into a Clark-notation qualified tag name for lxml. For 

59 example, ``qn('p:cSld')`` returns ``'{http://schemas.../main}cSld'``. 

60 Source: https://github.com/python-openxml/python-docx/ 

61 """ 

62 prefix, tagroot = tag.split(':') 

63 uri = nsmap[prefix] 

64 return '{{{}}}{}'.format(uri, tagroot)

67def xml2text(xml, pptx=False): 

68 """ 

69 A string representing the textual content of this run, with content 

70 child elements like ``<w:tab/>`` translated to their Python 

71 equivalent. 

72 Adapted from: https://github.com/python-openxml/python-docx/ 

73 """ 

74 text = u'' 

75 root = ET.fromstring(xml) 

76 if pptx is False: 

77 for child in root.iter(): 

78 if child.tag == qn('w:t'): 

79 t_text = child.text 

80 text += t_text if t_text is not None else '' 

81 elif child.tag == qn('w:tab'): 

82 text += '\t' 

83 elif child.tag in (qn('w:br'), qn('w:cr')): 

84 text += '\n' 

85 elif child.tag == qn("w:p"): 

86 text += '\n\n' 

87 else: 

88 for child in root.iter(): 

89 if child.tag == qn('a:t'): 

90 t_text = child.text 

91 text += t_text if t_text is not None else '' 

92 elif child.tag == qn('a:tab'): 92 ↛ 93line 92 didn't jump to line 93, because the condition on line 92 was never true

93 text += '\t' 

94 elif child.tag in (qn('a:br'), qn('a:cr')): 

95 text += '\n' 

96 elif child.tag in (qn("a:p"), qn("a:bodyPr"), qn("a:fld")): 

97 text += '\n\n' 

98 return text

101def process(docx, pptx=False, img_dir=None): 

102 text = u'' 

103 

104 # unzip the docx in memory 

105 zipf = zipfile.ZipFile(docx) 

106 filelist = zipf.namelist() 

107 images = [] 

108 page_count = 0 

109 

110 # get header text 

111 # there can be 3 header files in the zip 

112 header_xmls = 'ppt/header[0-9]*.xml' if pptx else 'word/header[0-9]*.xml' 

113 for fname in filelist: 

114 if re.match(header_xmls, fname): 114 ↛ 115line 114 didn't jump to line 115, because the condition on line 114 was never true

115 text += xml2text(zipf.read(fname)) 

116 

117 # get main text 

118 doc_xml = 'ppt/slides/slide[0-9]*.xml' if pptx else 'word/document.xml' 

119 if pptx: 

120 for fname in filelist: 

121 if re.match(doc_xml, fname): 

122 text += xml2text(zipf.read(fname), pptx=pptx) 

123 # add page count if pptx 

124 page_count += 1 

125 else: 

126 text += xml2text(zipf.read(doc_xml)) 

127 # get page count for docx 

128 page_count = get_pages_in_docx(docx) 

129 

130 # get footer text 

131 # there can be 3 footer files in the zip 

132 footer_xmls = 'ppt/footer[0-9]*.xml' if pptx else 'word/footer[0-9]*.xml' 

133 for fname in filelist: 

134 if re.match(footer_xmls, fname): 

135 text += xml2text(zipf.read(fname)) 

136 

137 for fname in filelist: 

138 _, extension = os.path.splitext(fname) 

139 if extension in [".jpg", ".jpeg", ".png", ".bmp"]: 

140 if img_dir: 140 ↛ 141line 140 didn't jump to line 141, because the condition on line 140 was never true

141 dst_fname = os.path.join(img_dir, os.path.basename(fname)) 

142 with open(dst_fname, "wb") as dst_f: 

143 dst_f.write(zipf.read(fname)) 

144 images.append(dst_fname) 

145 else: 

146 dst_f = tempfile.NamedTemporaryFile(dir=settings.TEMP_DIR) 

147 dst_f.write(zipf.read(fname)) 

148 images.append(dst_f) 

149 

150 zipf.close() 

151 

152 return text.strip(), images, page_count
