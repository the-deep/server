394def make_colormap(seq): 

395 """Return a LinearSegmentedColormap 

396 seq: a sequence of floats and RGB-tuples. The floats should be increasing 

397 and in the interval (0,1). 

398 """ 

399 seq = [(None,) * 3, 0.0] + list(seq) + [1.0, (None,) * 3] 

400 cdict = {'red': [], 'green': [], 'blue': []} 

401 for i, item in enumerate(seq): 

402 if isinstance(item, float): 

403 r1, g1, b1 = seq[i - 1] 

404 r2, g2, b2 = seq[i + 1] 

405 cdict['red'].append([item, r1, r2]) 

406 cdict['green'].append([item, g1, g2])

408 return mp.colors.LinearSegmentedColormap('CustomMap', cdict)
