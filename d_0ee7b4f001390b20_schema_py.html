33def get_lead_qs(info): 

34 lead_qs = Lead.objects.filter(project=info.context.active_project) 

35 # Generate querset according to permission 

36 if PP.check_permission(info, PP.Permission.VIEW_ALL_LEAD): 

37 return lead_qs 

38 elif PP.check_permission(info, PP.Permission.VIEW_ONLY_UNPROTECTED_LEAD): 

39 return lead_qs.filter(confidentiality=Lead.Confidentiality.UNPROTECTED) 

40 return Lead.objects.none()
