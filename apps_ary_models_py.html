156 def get_children_list(self): 

157 """ 

158 Returns list of nodes 

159 Each item is a dict consisting parents and node id 

160 Example return: [

163 {'title': 'All - Affected', 'parents': ['Affected', 'All'], 'id': 1}, 

164 ] 

165 """ 

166 # TODO: cache, but very careful 

167 nodes_list = [ 

168 { 

169 'title': self.title, 

170 'parents': [self.title], # includes self as well 

171 'id': self.id 

172 } 

173 ] 

174 children = self.children.all() 

175 if not children: 

176 return nodes_list

405 def get_actual_value(schema, value): 

406 value_function = FIELDS_KEYS_VALUE_EXTRACTORS.get(schema['name'], identity) 

407 if schema['type'] == Field.SELECT: 

408 # value should not be list but just in case it is a list

698 actual_value = schema['options'].get(value, value)

711 if not raw_data: 

712 return {} 

713 

714 if 'id' in schema: 

715 key = str(schema['id']) 

716 value = raw_data.get(key, '') 

717 return { 

718 'schema': schema,

721 } 

722 if isinstance(schema, dict): 

723 data = { 

724 k: Assessment.get_data_from_schema(v, raw_data) 

725 for k, v in schema.items() 

726 } 

727 elif isinstance(schema, list): 

728 data = [Assessment.get_data_from_schema(x, raw_data)
